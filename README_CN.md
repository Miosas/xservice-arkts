# xservice-arkts

一个用于生成基于xservice_kit的Flutter插件的工具，支持ArkTS/OHOS平台。

## 功能

- 为多个平台生成Flutter插件代码：
  - Android (Java)
  - iOS (Objective-C)
  - Dart
  - ArkTS/OHOS (HarmonyOS)

## 安装

```bash
npm install -g xservice-arkts
```

## 使用

```bash
xservice-arkts <目录> [选项]
```

### 选项

- `-o, --output <output>`: 保存输出的目录
- `-t, --type <type>`: 输入文件类型，支持json和yaml，默认为json
- `-p, --package <package>`: Java包名
- `-a, --arkts`: 为OHOS生成ArkTS代码

### 示例

```bash
xservice-arkts ./services -o ./gen -p com.xservice_kit.demo -t yaml --arkts
```

## 输入文件格式

该工具支持JSON和YAML格式的服务定义。

### YAML文件示例

```yaml
name: TestService
messages:
  - name: getTestData
    messageType: flutter
    returnType: String
    args:
      - name: userId
        type: String
  - name: updateTestData
    messageType: native
    returnType: bool
    args:
      - name: data
        type: Map
```

## 输出

该工具生成以下目录结构：

```
output/
  ├── java/           # Android代码
  ├── oc/             # iOS代码
  ├── dart/           # Flutter代码
  └── arkts/          # ArkTS/OHOS代码 (如果指定了--arkts)
```

## 生成的ArkTS代码示例

### 服务文件
```typescript
/**
 * Generated by code generator
 * Service: TestService
 * Package: com.xservice_kit.demo
 */

import { ServiceTemplate, ServiceGateway, MessageResult, MessageValue } from 'xservice_kit';
import { MethodResult } from '@ohos/flutter_ohos';

export class TestService {
  private static readonly mService: ServiceTemplate = new ServiceTemplate('TestService');

  public static getService(): ServiceTemplate {
    return TestService.mService;
  }

  public static register(): void {
    ServiceGateway.sharedInstance().addService(TestService.mService);
  }

  public static updateTestData(result: MessageResult<MessageValue>, data: Object): void {
    const args: Record<string, MessageValue> = {};
    args['data'] = data;
    const methodResult: MethodResult = {
      success: (o: Object): void => {
        if (o !== null && o !== undefined) {
          result.success(o as MessageValue);
        } else {
          result.error('return type error', '', '');
        }
      },
      error: (errorCode: string, errorMessage: string, errorDetails: Object): void => {
        if (result) {
          result.error(errorCode, errorMessage, errorDetails);
        }
      },
      notImplemented: (): void => {
        if (result) {
          result.notImplemented();
        }
      }
    };
    TestService.mService.invoke(
      'updateTestData',
      args,
      TestService.mService.methodChannelName(),
      methodResult
    );
  }
}
```
